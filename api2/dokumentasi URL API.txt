-- DOKUMENTASI URL API --

LOGIN
url	: localhost:8080/login
method	: POST
input	: username, password (JSON)
output	: status, username, message (JSON)

READ
url	: localhost:8080/nasabah
method	: GET
output	: List<nasabah> : id_nasabah, email, username, nama_lengkap, password, no_ktp, tgl_lahir, alamat, kode_rahasia, no_rek, jml_saldo, kode_cabang, created(JSON)

READ-ONE
url	: localhost:8080/nasabah/{id}
method	: GET
input	: id (param)
output	: nasabah : id_nasabah, email, username, nama_lengkap, password, no_ktp, tgl_lahir, alamat, kode_rahasia, no_rek, jml_saldo, kode_cabang, created(JSON)


BUKA REKENING (CREATE NASABAH)
url	: localhost:8080/nasabah
method	: POST
input	: nama_lengkap, email, password, no_ktp, tgl_lahir, alamat, kode_rahasia (JSON)
output	: message, status, username, id_nsb(JSON)


READ MUTASI
url	: localhost:8080/mutasi/{id}
method	: GET
input	: id (param)
output	: JSON Array 
	respon : id_nasabah, tgl_awal, tgl_akhir
	result : List<mutasi> : kode_transaksi, no_rek, tgl_trans, tujuan, jenis, keterangan, nominal 


READ HISTORY
url	: localhost:8080/history/{id}/{tgl_awal}/{tgl_akhir}
	contoh format tanggal : 		localhost:8080/history/2/2018-08-15/2018-09-13
method	: GET
input	: id, tgl_awal, tgl_akhir (param)
output	: JSON Array 
	respon : id_nasabah, tgl_awal, tgl_akhir
	result : List<history> : kode_transaksi, tgl_trans, tujuan, keterangan, nominal, status (JSON)


UPDATE PASSWORD
url	: localhost:8080/update_password
method	: POST
input	: id_nasabah, passwordl, passwordb1, passwordb2 (JSON)
output	: message, status (JSON)


UPDATE CODE
url	: localhost:8080/update_kode_rahasia
method	: POST
input	: id_nasabah, kode_rahasiaL, krb1, krb2 (JSON)
output	: message, status (JSON)


TRANSFER (CREATE)
url	: localhost:8080/transfer
method	: POST
input	: norek, id_nsb, nominal, uname, ket, kode_rhs (JSON)
output	: status, message (JSON)


TRANSFER (CEK NO REK)
url	: localhost:8080/ceknorek
method	: POST
input	: norek, id_nsb (JSON)
output	: status, message (JSON)


PEMBELIAN PULSA (CREATE)
url	: localhost:8080/pembelian
method	: POST
input	:nohp, id_nsb, provider, nominal, uname, kode_rhs (JSON)
output	:status, message (JSON)
